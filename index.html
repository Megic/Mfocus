	//插件编写采用js和样式分离的思想，Mfocus只处理js逻辑，因此给页面设计带来极高的灵活度。
	//所以在不同的设计中均可使用Mfocus处理常用的页面逻辑效果。
	//值得一提的是：Mfocus压缩后的大小不到2KB并且在多款浏览器中测试过，包括IE6。
	//Mfocus是作者在日常工作中为了拜托各种臃肿的jquery插件而设计的。

	//下面是相关默认参数 
	defaults = {      
   			 button: '.button',    //按钮父标签  可自定义
			 pre: '.pre', //向前 可自定义
			 next: '.next',//向后 可自定义
			 item: 'ul',//展示项父标签 可自定义
			 effect:'slide',//转换方式:slide/slideup/fade 三种切换形式
			 scroll:'true',//是否自动滚动 true/false
			 seamless:false,//是否无缝滚动
			 sctime:'4000',//滚动间隔 可自定义
			 moveef:'swing',//动画效果 swing/easeOutBounce/easeInOutExpo 内置三种seasing动画
			 speed:'500'  //动画时间 可自定义
			 };
